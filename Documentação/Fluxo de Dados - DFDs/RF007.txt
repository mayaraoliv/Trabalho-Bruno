dados-consulta-fornecedor =
    * Informações utilizadas pelo gerente para buscar fornecedores no sistema *
    nome-fornecedor + cnpj + {produto-fornecido}

nome-fornecedor =
    * Nome completo ou razão social do fornecedor *
    {caractere-válido}

telefone =
    * Número de telefone para contato do fornecedor *
    "(" + ddd + ")" + número-telefone

ddd =
    * Código de área do telefone *
    [0-9]{2}

número-telefone =
    * Número principal do telefone *
    [0-9]{8,9}

e-mail =
    * Endereço eletrônico do fornecedor *
    {caractere-válido} + "@" + domínio

domínio =
    * Parte final do e-mail que identifica o provedor *
    {caractere-válido} + "." + extensão

cnpj =
    * Cadastro Nacional da Pessoa Jurídica *
    [0-9]{14}

produto-fornecido =
    * Produto disponibilizado pelo fornecedor para a adega *
    código-produto + nome-produto + tipo + preço-fornecimento

código-produto =
    * Identificador único do produto fornecido *
    @id-produto

nome-produto =
    * Nome comercial do produto fornecido *
    {caractere-válido}

tipo =
    * Categoria ou tipo do produto fornecido *
    [“Bebida” | “Alimentício” | “Higiene” | “Outro”]

preço-fornecimento =
    * Valor pago ao fornecedor por unidade do produto *
    número-decimal

histórico-compras =
    * Registros de compras feitas com determinado fornecedor *
    {registro-compra-fornecedor}

registro-compra-fornecedor =
    * Cada compra registrada pelo sistema com o fornecedor *
    código-compra + data-compra + valor-total + {produto-fornecido}

data-compra =
    * Data em que a compra foi realizada *
    dd + "/" + mm + "/" + aaaa

valor-total =
    * Valor total da compra feita ao fornecedor *
    número-decimal

resultado-consulta =
    * Dados retornados ao gerente após pesquisa de fornecedores *
    nome-fornecedor + telefone + e-mail + cnpj + {produto-fornecido} + histórico-compras

Histórico de fornecedores =
    * Base de dados com todas as informações de fornecedores cadastrados *
    {@id-fornecedor + nome-fornecedor + telefone + e-mail + cnpj + {produto-fornecido} + histórico-compras}

número-decimal =
    [0-9]+("." [0-9]+)

caractere-válido =
    [A-Z | a-z | 0-9 | _ | . | - | espaço]
