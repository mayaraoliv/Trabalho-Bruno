dados-compra =
    * Informações registradas pelo atendente no momento da compra *
    nome-cliente + telefone + {item-compra} + forma-pagamento

item-compra =
    * Produto incluído na compra *
    código-produto + quantidade + preço-unitário

nome-cliente =
    * Nome completo do cliente que realiza a compra *
    {caractere-válido}

telefone =
    * Número de telefone do cliente para contato *
    "(" + ddd + ")" + número-telefone

ddd =
    * Código de área do telefone *
    [0-9]{2}

número-telefone =
    * Número principal do telefone *
    [0-9]{8,9}

código-produto =
    * Identificador único do produto comprado *
    @id-produto

quantidade =
    * Número de unidades adquiridas do produto *
    [0-9]+

preço-unitário =
    * Valor monetário de cada unidade do produto *
    número-decimal

forma-pagamento =
    * Método utilizado pelo cliente para pagamento da compra *
    [“Dinheiro” | “Cartão” | “Pix” | “Outro”]

valor-total =
    * Valor total da compra calculado pelo sistema *
    número-decimal

registro-compra =
    * Conjunto de informações registradas no banco de dados após a compra *
    @id-compra + nome-cliente + telefone + {item-compra} + forma-pagamento + valor-total

Vendas registradas =
    * Armazenamento que contém todos os registros de vendas realizadas *
    {@id-compra + nome-cliente + telefone + {item-compra} + forma-pagamento + valor-total}

Estoque =
    * Base de dados responsável por armazenar as informações de produtos disponíveis e atualizar as quantidades após cada compra *
    {@id-produto + nome-produto + tipo + marca + validade + quantidade + preço-unitário}

atualização-estoque =
    * Fluxo de dados que representa a atualização automática da quantidade de produtos no estoque após a compra *
    código-produto + quantidade-atualizada

quantidade-atualizada =
    * Novo valor de quantidade do produto após o registro da compra *
    [0-9]+

número-decimal =
    [0-9]+("." [0-9]+)

caractere-válido =
    [A-Z | a-z | 0-9 | _ | . | - | espaço]
